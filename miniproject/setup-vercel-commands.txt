// Copy and run these commands one by one in your terminal:

vercel env add NEXT_PUBLIC_SUPABASE_URL production
// Enter when prompted: https://your-project-id.supabase.co

vercel env add NEXT_PUBLIC_SUPABASE_ANON_KEY production
// Enter when prompted: your-supabase-anon-key

vercel env add SUPABASE_SERVICE_ROLE_KEY production
// Enter when prompted: your-supabase-service-role-key

vercel env add DATABASE_URL production
// Enter when prompted: postgresql://postgres:your-password@db.your-project-id.supabase.co:5432/postgres

vercel env add JWT_SECRET production
// Enter when prompted: your-secure-jwt-secret

vercel env add HUGGINGFACE_API_KEY production
// Enter when prompted: your-huggingface-api-key

vercel env add COHERE_API_KEY production
// Enter when prompted: your-cohere-api-key

vercel env add DEFAULT_AI_PROVIDER production
// Enter when prompted: huggingface

vercel env add NEXT_PUBLIC_GOOGLE_MAPS_API_KEY production
// Enter when prompted: your-google-maps-api-key

vercel env add NEXT_PUBLIC_APP_URL production
// Enter when prompted: https://your-actual-vercel-domain.vercel.app

// After setting all variables, redeploy:
vercel --prod
